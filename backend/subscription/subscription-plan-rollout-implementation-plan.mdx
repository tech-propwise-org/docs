```typescript
export class EntityCapService {
  async getEntityCount(orgId: string, entityType: EntityType): Promise<number> {
    const repository = this.getRepositoryForEntity(entityType);
    
    const whereClause: any = {
      organizationId: orgId,
```

<Step title="Consumption Routing">
Implement org-first, then personal consumption model:

```typescript
export class CreditConsumptionService {
  async consumeCredits(context: SpendContext, amount: number): Promise<ConsumeResult> {
    const { userId, orgId, source } = context;
    
    // Determine consumption path
    if (this.isOrgLevelConsumption(source)) {
      return this.consumeFromOrgPool(orgId, amount);
    }
    
    // User-level consumption: org allowance first, then personal
    const userCeiling = await this.creditMeteringService.getUserCreditCeiling(userId);
    const userOrgSpent = await this.getUserOrgSpending(userId, orgId);
    const availableFromOrg = Math.max(0, userCeiling - userOrgSpent);
    
    let remaining = amount;
    let consumedFromOrg = 0;
    let consumedFromPersonal = 0;
    
    if (availableFromOrg > 0) {
      consumedFromOrg = Math.min(remaining, availableFromOrg);
      await this.consumeFromOrgPool(orgId, consumedFromOrg);
      remaining -= consumedFromOrg;
    }
    
    if (remaining > 0) {
      consumedFromPersonal = await this.consumeFromPersonalCredits(userId, remaining);
    }
    
    return {
      totalConsumed: consumedFromOrg + consumedFromPersonal,
      fromOrg: consumedFromOrg,
      fromPersonal: consumedFromPersonal,
      success: (consumedFromOrg + consumedFromPersonal) >= amount
    };
  }
}
```
</Step>

<Step title="OpenAI Cost Mapping">

```typescript
export class CostMappingService {
  private readonly COST_MAP = {
    'gpt-4': { input: 0.03, output: 0.06 }, // Per 1K tokens
    'gpt-4-turbo': { input: 0.01, output: 0.03 },
    'gpt-3.5-turbo': { input: 0.0015, output: 0.002 }
  };
  
  calculateCreditCost(model: string, inputTokens: number, outputTokens: number): number {
    const rates = this.COST_MAP[model] || this.COST_MAP['gpt-4'];
    const usdCost = (inputTokens / 1000) * rates.input + (outputTokens / 1000) * rates.output;
    
    // Convert USD to credits (implementation depends on credit-to-USD ratio)
    return Math.ceil(usdCost * this.CREDITS_PER_USD);
  }
}
```
