This document provides the complete implementation plan for rolling out the new Free/Pro/Business subscription packaging in AED, implementing the strategy outlined in `SUBSCRIPTION_PACKAGING_STRATEGY.md`.

<Note>
This plan addresses 12 identified gaps from the strategy document and groups them into sequential phases, prioritizing backend implementation before frontend changes.
</Note>

## Overview

The rollout implements a three-tier subscription model with:
- **Free**: 20 AI credits/month, 10 contacts/deals/companies, 1GB storage
- **Pro**: 2000 AI credits/month, unlimited entities, 10GB storage, 90-day trial
- **Business**: 4000 AI credits/month, unlimited entities, 25GB storage, 90-day trial

<Warning>
Several existing systems require replacement rather than extension. The AI budget gate is already live and must be behavior-equivalent during transition.
</Warning>

export class CreditMeteringService {
  async getOrgCreditBalance(orgId: string): Promise<number> {

  async getUserCreditAllocation(userId: string): Promise<UserAllocation> {

<Step title="Implement spend context routing">

interface SpendContext {
  actorId: string;
  organizationId: string;
  channel: 'org' | 'personal';
  feature: 'assistant' | 'agent' | 'valuation';
}

async deductCredits(context: SpendContext, amount: number): Promise<void> {
  if (context.channel === 'org') {
    await this.deductFromOrgPool(context.organizationId, amount);
  } else {
    // Try org allocation first, then personal
    const userAllocation = await this.getUserOrgAllocation(context.actorId);
    if (userAllocation >= amount) {
      await this.deductFromUserAllocation(context.actorId, amount);
    } else {
      await this.deductFromPersonalCredits(context.actorId, amount);
    }
  }
}
