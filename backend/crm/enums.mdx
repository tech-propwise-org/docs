This page lists all enum types used across the CRM module.

## Person and contact enums

```sql
-- Contact method preference
CREATE TYPE contact_method AS ENUM (
  'email', 'phone', 'whatsapp', 'sms', 'in_person'
);

-- Contact status in CRM workflow
CREATE TYPE contact_status AS ENUM (
  'active', 'inactive', 'churned', 'blocked'
);

-- Language preference
CREATE TYPE preferred_language AS ENUM (
  'en', 'ar', 'fr', 'es', 'de', 'zh', 'hi', 'pt', 'ru', 'ja'
);

-- Original source tracking
CREATE TYPE original_source AS ENUM (
  'website', 'social_media', 'referral', 'advertisement',
  'cold_call', 'walk_in', 'event', 'partner', 'other'
);
```

## Channel communication enums

```sql
-- Communication channel types
CREATE TYPE channel_type AS ENUM (
  'whatsapp', 'instagram', 'facebook', 'telegram', 'sms', 'webchat'
);

-- Channel connection status
CREATE TYPE channel_status AS ENUM (
  'active', 'inactive', 'blocked', 'unsubscribed'
);
```

## Company enums

```sql
-- Company classification
CREATE TYPE company_type AS ENUM (
  'customer', 'vendor', 'partner', 'competitor', 'other'
);

-- Company size categories
CREATE TYPE company_size AS ENUM (
  'startup', 'small', 'medium', 'large', 'enterprise'
);
```

## Notes and tags enums

```sql
-- Note categorization
CREATE TYPE note_type AS ENUM (
  'general', 'meeting', 'call', 'follow_up', 'internal', 'system'
);

-- Note access control
CREATE TYPE note_visibility AS ENUM (
  'public', 'private', 'team_only'
);

-- Tag classification
CREATE TYPE tag_type AS ENUM ('system', 'custom');
```

## Stage enums

```sql
-- System lead stage type
CREATE TYPE system_lead_stage_type AS ENUM (
  'new', 'contacted', 'qualified', 'converted', 'disqualified'
);

-- System deal stage type
CREATE TYPE system_deal_stage_type AS ENUM (
  'new', 'proposal', 'negotiation', 'closed_won', 'closed_lost'
);
```

## Stakeholder enums

```sql
-- Stakeholder entity type
CREATE TYPE stakeholder_entity_type AS ENUM ('lead', 'deal');

-- Access level for stakeholders
CREATE TYPE access_level AS ENUM ('read', 'write', 'full');

-- Stakeholder history action
CREATE TYPE stakeholder_action AS ENUM (
  'added', 'removed', 'commission_changed',
  'transferred', 'inherited', 'primary_changed'
);

-- Transfer status
CREATE TYPE transfer_status AS ENUM ('pending', 'approved', 'rejected', 'cancelled');

-- Payment status
CREATE TYPE payment_status AS ENUM ('pending', 'approved', 'paid', 'cancelled', 'on_hold');
```

## Activity and communication enums

```sql
-- Activity target type
CREATE TYPE activity_target_type AS ENUM ('lead', 'deal', 'contact', 'company');

-- Activity type
CREATE TYPE activity_type AS ENUM (
  'call', 'email', 'sms', 'whatsapp', 'meeting',
  'site_visit', 'note', 'task', 'other'
);

-- Lead intent (single value, NOT array)
CREATE TYPE lead_intent AS ENUM (
  'buyer', 'seller', 'renter', 'landlord', 'investor'
);

-- Deal contact role
CREATE TYPE deal_contact_role AS ENUM (
  'buyer', 'seller', 'tenant', 'landlord', 'investor',
  'guarantor', 'decision_maker', 'influencer', 'other'
);
```

## Channel table enums

```sql
-- Task status
CREATE TYPE task_status AS ENUM (
  'pending', 'in_progress', 'completed', 'cancelled', 'deferred'
);

-- Meeting type
CREATE TYPE meeting_type AS ENUM ('in_person', 'video_call', 'phone_call');

-- Meeting status
CREATE TYPE meeting_status AS ENUM (
  'scheduled', 'confirmed', 'in_progress', 'completed',
  'cancelled', 'no_show', 'rescheduled'
);

-- SMS direction / status
CREATE TYPE sms_direction AS ENUM ('inbound', 'outbound');
CREATE TYPE sms_status AS ENUM ('pending', 'sent', 'delivered', 'failed');

-- Site visit status
CREATE TYPE site_visit_status AS ENUM (
  'scheduled', 'confirmed', 'in_progress', 'completed',
  'cancelled', 'no_show'
);
```
